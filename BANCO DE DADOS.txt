--SCRIT DE CRIAÇÃO DA TABELA PRODUTO

CREATE TABLE invest_portf.TB_PRODUTO(
ID_PRODUTO INT AUTO_INCREMENT NOT NULL,
DS_ATIVO VARCHAR(100) NOT NULL,
DS_TIPO_PRODUTO VARCHAR(100) NOT NULL,
DS_ESPECIFICACAO_TITULO VARCHAR(100) NOT NULL,
DS_NEGOCIACAO VARCHAR(100) NOT NULL,
NR_QUANTIDADE_DISPONIVEL INT NOT NULL,
VL_PRECO DECIMAL(11,2) NOT NULL,
DS_EMAIL_ADMINISTRADOR VARCHAR(100) NOT NULL,
DT_ALTERACAO DATETIME NOT NULL,
DT_VENCIMENTO DATETIME NOT NULL,
CONSTRAINT PK_PRODUTO PRIMARY KEY (ID_PRODUTO)
);

--SCRIT DE CRIAÇÃO DA TABELA ORDEM

CREATE TABLE invest_portf.TB_ORDEM(
NR_NOTA INT AUTO_INCREMENT NOT NULL,
ID_PRODUTO INT NOT NULL,
NR_CPF_CNPJ BIGINT NOT NULL,
DS_ATIVO VARCHAR(100) NOT NULL,
DS_TIPO_PRODUTO VARCHAR(100) NOT NULL,
DS_NEGOCIACAO VARCHAR(100) NOT NULL,
DS_TIPO_OPERACAO VARCHAR(100) NOT NULL,
DS_ESPECIFICACAO_TITULO VARCHAR(100) NOT NULL,
NR_QUANTIDADE INT NOT NULL,
VL_PRECO DECIMAL(11,2) NOT NULL,
VL_OPERACAO DECIMAL(11,2) NOT NULL,
DT_OPERACAO DATETIME,
CONSTRAINT PK_ORDEM PRIMARY KEY (NR_NOTA),
CONSTRAINT fk_PRODUTO_ORDEM FOREIGN KEY (ID_PRODUTO) REFERENCES invest_portf.TB_PRODUTO (ID_PRODUTO)
);













--SCRIT DE CRIAÇÃO DA TABELA CARTEIRA

CREATE TABLE invest_portf.TB_CARTEIRA(
ID_CARTEIRA INT AUTO_INCREMENT NOT NULL,
ID_PRODUTO INT NOT NULL,
NR_CPF_CNPJ INT LONG NOT NULL,
DS_ATIVO VARCHAR(100) NOT NULL,
DS_TIPO_PRODUTO VARCHAR(100) NOT NULL,
NR_QUANTIDADE INT NOT NULL,
VL_POSICAO DECIMAL(11,2) NOT NULL,
DT_ALTERACAO DATETIME,
CONSTRAINT PK_CARTEIRA PRIMARY KEY (ID_CARTEIRA),
CONSTRAINT fk_PRODUTO_CARTEIRA FOREIGN KEY (ID_PRODUTO) REFERENCES invest_portf.TB_PRODUTO (ID_PRODUTO)
);
